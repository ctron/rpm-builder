011package de.dentrassi.rpm.builder;
012
013import java.io.File;
014
015public class PackageEntry extends EntryDetails
016{
017    public static class Collector
018    {
019        private File from;
020
021        private boolean directories = true;
022
023        private boolean symbolicLinks = false;
024
025        private String[] includes;
026
027        private String[] excludes;
028
029        public File getFrom ()
030        {
031            return this.from;
032        }
033
034        public void setFrom ( final File from )
035        {
036            this.from = from;
037        }
038
039        public void setDirectories ( final boolean directories )
040        {
041            this.directories = directories;
042        }
043
044        public boolean isDirectories ()
045        {
046            return this.directories;
047        }
048
049        public void setSymbolicLinks ( final boolean symbolicLinks )
050        {
051            this.symbolicLinks = symbolicLinks;
052        }
053
054        public boolean isSymbolicLinks ()
055        {
056            return this.symbolicLinks;
057        }
058
059        public void setIncludes ( final String[] includes )
060        {
061            this.includes = includes;
062        }
063
064        public String[] getIncludes ()
065        {
066            return this.includes;
067        }
068
069        public void setExcludes ( final String[] excludes )
070        {
071            this.excludes = excludes;
072        }
073
074        public String[] getExcludes ()
075        {
076            return this.excludes;
077        }
078
079        @Override
080        public String toString ()
081        {
082            return String.format ( "[collector - from: %s,  directories: %s, symLinks: %s, includes: %s, excludes: %s]", this.from, this.directories, this.symbolicLinks, this.includes, this.excludes );
083        }
084    }
085
086    private String name;
087
088    private Boolean directory;
089
090    private File file;
091
092    private Collector collect;
093
094    private String linkTo;
095
096    private String ruleset;
097
098    public String getName ()
099    {
100        return this.name;
101    }
102
103    public void setName ( final String name )
104    {
105        this.name = name;
106    }
107
108    public Boolean getDirectory ()
109    {
110        return this.directory;
111    }
112
113    public void setDirectory ( final Boolean directory )
114    {
115        this.directory = directory;
116    }
117
118    public File getFile ()
119    {
120        return this.file;
121    }
122
123    public void setFile ( final File file )
124    {
125        this.file = file;
126    }
127
128    public Collector getCollect ()
129    {
130        return this.collect;
131    }
132
133    public void setCollect ( final Collector collect )
134    {
135        this.collect = collect;
136    }
137
138    public void setLinkTo ( final String linkTo )
139    {
140        this.linkTo = linkTo;
141    }
142
143    public String getLinkTo ()
144    {
145        return this.linkTo;
146    }
147
148    public void setRuleset ( final String ruleset )
149    {
150        this.ruleset = ruleset;
151    }
152
153    public String getRuleset ()
154    {
155        return this.ruleset;
156    }
157
158    @Override
159    public void validate ()
160    {
161        if ( this.name == null || this.name.isEmpty () )
162        {
163            throw new IllegalStateException ( "'name' must not be empty" );
164        }
165
166        int sources = 0;
167
168        sources += this.directory != null && this.directory ? 1 : 0;
169        sources += this.file != null ? 1 : 0;
170        sources += this.collect != null ? 1 : 0;
171        sources += this.linkTo != null ? 1 : 0;
172
173        if ( sources != 1 )
174        {
175            throw new IllegalStateException ( "Exactly one of 'file', 'directory', 'linkTo' or 'collect' must be specified." );
176        }
177
178        super.validate ();
179    }
180}
