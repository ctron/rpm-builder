012package de.dentrassi.rpm.builder;

014import java.util.EnumSet;
015import java.util.Set;
016
017import org.eclipse.packagedrone.utils.rpm.FileFlags;
018import org.eclipse.packagedrone.utils.rpm.build.FileInformation;
019
020public class EntryDetails
021{
022    private Short mode;
023
024    private Boolean configuration;
025
026    private Boolean documentation;
027
028    private Boolean license;
029
030    private Boolean readme;
031
032    private Boolean ghost;
033
034    private Boolean missingOk;
035
036    private Boolean noreplace;
037
038    private String user;
039
040    private String group;
041
042    public void setMode ( final String mode )
043    {
044        this.mode = Short.parseShort ( mode, 8 );
045    }
046
047    public Short getMode ()
048    {
049        return this.mode;
050    }
051
052    public void setConfiguration ( final Boolean configuration )
053    {
054        this.configuration = configuration;
055    }
056
057    public Boolean getConfiguration ()
058    {
059        return this.configuration;
060    }
061
062    public void setDocumentation ( final Boolean documentation )
063    {
064        this.documentation = documentation;
065    }
066
067    public Boolean getDocumentation ()
068    {
069        return this.documentation;
070    }
071
072    public void setLicense ( final Boolean license )
073    {
074        this.license = license;
075    }
076
077    public Boolean getLicense ()
078    {
079        return this.license;
080    }
081
082    public void setReadme ( final Boolean readme )
083    {
084        this.readme = readme;
085    }
086
087    public Boolean getReadme ()
088    {
089        return this.readme;
090    }
091
092    public void setGhost ( final Boolean ghost )
093    {
094        this.ghost = ghost;
095    }
096
097    public Boolean getGhost ()
098    {
099        return this.ghost;
100    }
101
102    public void setMissingOk ( final Boolean missingOk )
103    {
104        this.missingOk = missingOk;
105    }
106
107    public Boolean getMissingOk ()
108    {
109        return this.missingOk;
110    }
111
112    public void setNoreplace ( final Boolean noreplace )
113    {
114        this.noreplace = noreplace;
115    }
116
117    public Boolean getNoreplace ()
118    {
119        return this.noreplace;
120    }
121
122    public void setUser ( final String user )
123    {
124        this.user = user;
125    }
126
127    public String getUser ()
128    {
129        return this.user;
130    }
131
132    public void setGroup ( final String group )
133    {
134        this.group = group;
135    }
136
137    public String getGroup ()
138    {
139        return this.group;
140    }
141
142    public void validate ()
143    {
144    }
145
146    public boolean apply ( final FileInformation info )
147    {
148        boolean didApply = false;
149
150        if ( this.configuration != null )
151        {
152            setFlag ( info, FileFlags.CONFIGURATION );
153            didApply = true;
154        }
155        if ( this.documentation != null )
156        {
157            setFlag ( info, FileFlags.DOC );
158            didApply = true;
159        }
160        if ( this.license != null )
161        {
162            setFlag ( info, FileFlags.LICENSE );
163            didApply = true;
164        }
165        if ( this.readme != null )
166        {
167            setFlag ( info, FileFlags.README );
168            didApply = true;
169        }
170        if ( this.ghost != null )
171        {
172            setFlag ( info, FileFlags.GHOST );
173            didApply = true;
174        }
175        if ( this.missingOk != null )
176        {
177            setFlag ( info, FileFlags.MISSINGOK );
178            didApply = true;
179        }
180        if ( this.noreplace != null )
181        {
182            setFlag ( info, FileFlags.NOREPLACE );
183            didApply = true;
184        }
185        if ( this.user != null && !this.user.isEmpty () )
186        {
187            info.setUser ( this.user );
188            didApply = true;
189        }
190        if ( this.group != null && !this.group.isEmpty () )
191        {
192            info.setGroup ( this.group );
193            didApply = true;
194        }
195        if ( this.mode != null )
196        {
197            info.setMode ( this.mode );
198            didApply = true;
199        }
200        return didApply;
201    }
202
203    private void setFlag ( final FileInformation info, final FileFlags flag )
204    {
205        final Set<FileFlags> flags = info.getFileFlags ();
206        if ( flags == null )
207        {
208            info.setFileFlags ( EnumSet.of ( flag ) );
209        }
210        else
211        {
212            flags.add ( flag );
213        }
214    }
