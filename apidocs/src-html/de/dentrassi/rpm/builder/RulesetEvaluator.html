011package de.dentrassi.rpm.builder;
012
013import java.util.Collection;
014import java.util.LinkedHashMap;
015import java.util.LinkedHashSet;
016import java.util.List;
017import java.util.Map;
018import java.util.Set;
019import java.util.stream.Collectors;
020
021import org.eclipse.packagedrone.utils.rpm.build.FileInformation;
022import org.eclipse.packagedrone.utils.rpm.build.PayloadEntryType;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026public class RulesetEvaluator
027{
028    private static final Logger logger = LoggerFactory.getLogger ( RulesetEvaluator.class );
029
030    private class RulesetInstance
031    {
032        private final String id;
033
034        private final List<Rule> rules;
035
036        private final String parentRuleset;
037
038        public RulesetInstance ( final String id, final List<Rule> rules, final String parentRuleset )
039        {
040            this.id = id;
041            this.rules = rules;
042            this.parentRuleset = parentRuleset;
043        }
044
045        private void eval ( final Object object, final PayloadEntryType type, final String targetName, final FileInformation info, final Set<String> knownSets )
046        {
047            if ( knownSets.contains ( this.id ) )
048            {
049                throw new IllegalStateException ( String.format ( "Recursive calling of rulesets is not allowed- current: %s, previous: %s", this.id, knownSets.stream ().collect ( Collectors.joining ( ", " ) ) ) );
050            }
051
052            knownSets.add ( this.id );
053
054            for ( final Rule rule : this.rules )
055            {
056                logger.debug ( "Testing rule {}", rule );
057
058                if ( rule.matches ( object, type, targetName ) )
059                {
060                    logger.debug ( "  Rule matches. Applying information ..." );
061                    if ( rule.apply ( info ) )
062                    {
063                        logger.debug ( "    Information: {}", info );
064                    }
065                    if ( rule.isLast () )
066                    {
067                        logger.debug ( "  Last rule" );
068                        return;
069                    }
070                }
071            }
072
073            if ( this.parentRuleset != null && !this.parentRuleset.isEmpty () )
074            {
075                logger.debug ( "Running parent ruleset: '{}'", this.parentRuleset );
076                RulesetEvaluator.this.eval ( this.parentRuleset, object, type, targetName, info );
077            }
078        }
079
080        public void eval ( final Object object, final PayloadEntryType type, final String targetName, final FileInformation info )
081        {
082            eval ( object, type, targetName, info, new LinkedHashSet<> () );
083        }
084    }
085
086    private final Map<String, RulesetInstance> rulesets = new LinkedHashMap<> ();
087
088    public RulesetEvaluator ( final Collection<Ruleset> rulesets )
089    {
090        this ( null, rulesets );
091    }
092
093    public RulesetEvaluator ( final RulesetEvaluator parent, final Collection<Ruleset> rulesets )
094    {
095        for ( final Ruleset set : rulesets )
096        {
097            this.rulesets.put ( set.getId (), convert ( set ) );
098        }
099    }
100
101    private RulesetInstance convert ( final Ruleset set )
102    {
103        set.validate ();
104
105        return new RulesetInstance ( set.getId (), set.getRules (), set.getDefaultRuleset () );
106    }
107
108    public void eval ( final String ruleId, final Object object, final PayloadEntryType type, final String targetName, final FileInformation info )
109    {
110        final RulesetInstance ruleset = this.rulesets.get ( ruleId );
111
112        if ( ruleset == null )
113        {
114            throw new IllegalStateException ( String.format ( "Unknown rule: '%s'", ruleId ) );
115        }
116
117        ruleset.eval ( object, type, targetName, info );
118    }
119}
